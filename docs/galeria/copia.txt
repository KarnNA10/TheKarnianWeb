    function escInf(dataArray){
        // Obtener la colección de elementos <article>
        const articles = document.querySelectorAll('.galeria article');
    
        // Verificar si la cantidad de elementos en 'dataArray' coincide con la cantidad de elementos <article>
        if (dataArray.length !== articles.length) {
            console.error('La longitud de dataArray no coincide con la cantidad de elementos <article>.');
            return;
        }
    
        // Iterar sobre cada elemento <article> y asignar los datos del array
        articles.forEach((article, index) => {
            const data = dataArray[index];
    
            // Verificar si el array de datos tiene la longitud esperada
            if (data.length === 4) {
                const [imgDir, type, price, descr] = data;
    
                // Obtener los elementos dentro del <article>
                const imgElement = article.querySelector('.prodImg');
                const figCaptionName = article.querySelector('figcaption:not(.price)');
                const figCaptionPrice = article.querySelector('.price');
    
                // Asignar los valores
                imgElement.src = `/img/prod/${imgDir}`;
                figCaptionName.textContent = descr;
                figCaptionPrice.textContent = `$${price}`;
                // Asignar la función amp al evento onclick
                article.querySelector('a').onclick = function() {
                    amp(imgDir, type, price, descr);
                };
            } else {
                console.error('Cada array de datos debe tener exactamente 4 elementos.');
            }
        });
    }
